Configuraciones#

  • Crear una carpeta para el proyecto y dentro ejecutar npm init

  • RECOMENDADO Instalar nodemon: npm install nodemon y agregar al package.json

  • Instalar ExpressJS: npm install express --save

Archivo Inicial#

  • Se crea un archivo index.js o simil:

 1// importamos la librería express:
 2const express = require('express');
 3
 4// instanciamos express:
 5const app = express();
 6// definimos el puerto:
 7const port = 3000;
 8
 9// se preparan los datos hardcodeados:
10let consolas = [
11    {marca: 'Sony', modelo: 'PlayStation', lanzamiento: 1994},
12    {marca: 'Sega', modelo: 'MegaDrive ', lanzamiento: 1991},
13    {marca: 'Nintendo', modelo: 'Gameboy', lanzamiento: 1989},
14    {marca: 'Microsoft', modelo: 'XBOX', lanzamiento: 2001},
15]
16
17// Creamos una ruta raiz:
18app.get('/', (req, res)=>{
19    // se define la cabezera y el cuerpo:
20    res.status(200).send(consolas); // no hay que parsear a json.
21});
22
23// creamos el servidor:
24app.listen(port, ()=>{
25    console.log(`Servidor ejecutandose en http://localhost:${port}`);
26});

Habilitar CORS#

Por defecto el proyecto tiene las CORS restringidas, de modo que si vamos a conectarnos desde una aplicación externa se bloqueará.

  • Para trabajar con cors se instala npm install cors --save

CORS Publicas#

Habilitar cors para hacer publica la API:

1const express = require('express');
2
3const app = express();
4const port = 3500;
5// importamos la librería cors:
6const cors = require('cors');
7
8// decimos a express que defina los cors (por defecto ahora es pública):
9app.use(cors());

Ejemplo de Get con parámetros:

 1const express = require('express');
 2
 3const app = express();
 4const port = 3000;
 5
 6let consolas = [
 7    {marca: 'Sony', modelo: 'PlayStation', lanzamiento: 1994},
 8    {marca: 'Sega', modelo: 'MegaDrive ', lanzamiento: 1991},

16
17// petición get de una ruta con un parámetro:

Paso 1 - Iniciar proyecto#

  • Crear una carpeta para el proyecto.

  • Crear un repositorio en github, gitlab o simil.

  • ejecutar npm init para iniciar el proyecto.

  • instalar express: npm install express --save

Paso 2 - Preparar estructura#

  • crear archivos index.js y app.js

  • crear carpetas controllers, routes y models

Paso 10 - Crear controlador para gestionar peticiones#

en la carpeta controllers se crea un nuevo controlador llamado taskController.js:

  1const Task = require('../models/taskModel');
  2
  3// la funcion será asíncrona para que espere a terminar de guardar antes de proceder:
  4async function createTask(req, res){
  5    // se crea una instancia del modelo:
  6    const task = new Task();
  7    // recuperar los parámetros del body:
  8    const params = req.body;
  9
 10    // recuperar los datos del body y añadirlos al modelo:
 11    task.title = params.title;
 12    task.description = params.description;
 13
 14    // guardar datos:
 15    try{
 16        // guardar resultados en la base de datos:
 17        const taskStore = await task.save(); // el proceso será lineal hasta llegar al await (evita ejecutar todo a la vez)
 18
 19        // revisar si no se ha guardado la tarea:
 20        if(!taskStore){
 21            res.status(400).send({msg: "No se ha guardado la tarea"});
 22        }else{
 23            res.status(200).send({task: taskStore});
 24        }
 25    }catch (error){
 26        // devolver error 500 si sale mal:
 27        res.status(500).send(error);
 28    }
 29}
 30
 31// función para listar tareas:
 32async function getTasks(req, res){
 33    try{
 34        // recuperar tareas de base de datos:
 35        const tasks = await Task.find().sort({create_at: -1}); // opcional: ordenar resultados con sort por un parámetro seleccionado.
 36        // Agregar condición dentro de find: const tasks = await Task.find({completed: false}).sort({create_at: -1});
 37
 38        // comprobar si hay tareas:
 39        if(!tasks){
 40            res.status(400).send({msg: "Error al obtener las tareas"});
 41        }else {
 42            res.status(200).send(tasks);
 43        }
 44    }catch(error){
 45        res.status(500).send(error);
 46    }
 47}
 48
 49// recuperar un solo resultado:
 50async function getTask(req, res){
 51    // recuperar el id:
 52    const taskId = req.params.id;
 53
 54    try{
 55        // se hace un findById con el id recibido por request:
 56        const task = await Task.findById(taskId);
 57
 58        if(!task){
 59            res.status(400).send({msg: "No existe la tarea"});
 60        }else{
 61            res.status(200).send(task);
 62        }
 63    }catch(error){
 64        res.status(500).send(error);
 65    }
 66}
 67
 68// crear la función asincrona para actualizar tareas:
 69async function updateTask(req, res){
 70    // recuperar el id:
 71    const taskId = req.params.id;
 72    // recuperar parametros a actualizar:
 73    const params = req.body;
 74
 75    try{
 76        // realizar consulta para recuperar la tarea y le pasamos los datos a actualizar:
 77        const task = await Task.findByIdAndUpdate(taskId, params); // OJO: ahora usamos findByIdAndUpdate()
 78
 79
 80        if(!task){
 81            res.status(400).send({msg: "No existe la tarea"});
 82        }else{
 83            res.status(200).send(task);
 84        }
 85    }catch(error){
 86        res.status(500).send(error);
 87    }
 88}
 89
 90// función para eliminar datos:
 91async function deleteTask(req, res){
 92    // recuperar el id:
 93    const taskId = req.params.id;
 94
 95    try{
 96        // realizar delete:
 97        const task = await Task.findByIdAndDelete(taskId);
 98
 99        if(!task){
100            res.status(400).send({msg: "No existe la tarea"});
101        }else{
102            res.status(200).send({msg: "Se ha eliminado la tarea correctamente"});
103        }
104    }catch(error){
105        res.status(500).send(error);
106    }
107}
108
109module.exports = {
110    createTask,
111    getTasks,
112    getTask,
113    updateTask,
114    deleteTask
115}
